Positivo = []
Negativo = []

while True:
    Entrada = int(input())
    
    if Entrada > 0:
        Positivo.append(Entrada)
    elif Entrada < 0:
        Negativo.append(Entrada)
    else:
        break
    
for i in range (len(Negativo)):
    if i != (len(Negativo) - 1):
        print(Negativo[i], end = ', ')
    else:
        print(Negativo[i])
        
for i in range (len(Positivo)):
    if i != (len(Positivo) - 1):
        print(Positivo[i], end = ', ')
    else:
        print(Positivo[i])
