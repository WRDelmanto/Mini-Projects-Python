Entrada = int(input('Digite um número: '))
Quantidade_de_Digitos = 0
Soma_dos_Valores_Digitados = 0
Media_dos_Valores_Digitados = 0

while Entrada != 0:
    Quantidade_de_Digitos = Quantidade_de_Digitos + 1
    Soma_dos_Valores_Digitados = Soma_dos_Valores_Digitados + Entrada
    Entrada = int(input('Digite um número: '))
else:
    Media_dos_Valores_Digitados = Soma_dos_Valores_Digitados/Quantidade_de_Digitos
    print('Quantidade de Dígitos: %i' %Quantidade_de_Digitos)
    print('Soma dos valores digitado: %i' %Soma_dos_Valores_Digitados)
    print('Média dos valores digitado: %.2f' %Media_dos_Valores_Digitados)
