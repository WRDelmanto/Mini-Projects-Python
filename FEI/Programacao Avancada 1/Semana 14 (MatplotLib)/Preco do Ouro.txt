import matplotlib.pyplot as plt

import pandas



Date = []

Price = []



Gold_Prices = pandas.read_csv('gold_prices_monthly.csv')



Date = Gold_Prices['Date']

Price = Gold_Prices['Price']



fig, ax = plt.subplots() 



ax.plot(Date, Price)



ax.set_title('Preço Histórico do Ouro')

ax.legend(['Preço do Ouro'])

ax.set_xlabel('Ano')

ax.set_ylabel('Preço')



plt.show()
